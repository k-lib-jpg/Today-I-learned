#ローカルリポジトリの新規作成（.gitディレクトリが作成される）

git init

#.gitディレクトリとは

以下のファイルで構成されるディレクトリ
・リポジトリ（圧縮ファイル、ツリーファイル、コミットファイル）
・インデックスファイル
・設定ファイル

#ディレクトリ作成、ディレクトリ移動

mkdir (ファイル名)
cd (ファイルパス）

#gitリポジトリのコピーを作成する

git clone (githubのurl)

#変更をステージに追加する

git add .

#変更を記録する（コミット）

git commit
git commit -m "(推奨メッセージ：変更内容と変更理由を書く）"
git commit -v
